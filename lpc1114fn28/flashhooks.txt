\ For experimentation, there are three hooks to write Flash access code in Forth.
\ This is a simple demonstration on how to use them.

compiletoram

: flash-init  ( -- ) ."  Flash init" cr ;
: flash-h!    ( x addr ) ."  Flash write " hex. ." : " hex. cr ;
: flash-flush ( -- ) ."  Flash flush" cr ;

flash-init

  ' flash-init  hook-initflash !
  ' flash-h!    hook-hflash! !
  ' flash-flush hook-flushflash !
